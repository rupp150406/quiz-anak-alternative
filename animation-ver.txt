        let assignedCountries = JSON.parse(localStorage.getItem('assignedCountries')) || null;

        function playClick() { document.getElementById('clickSound').play().catch(()=>{}); }
        function playBack() { document.getElementById('backSound').play().catch(()=>{}); }

        function enableAudio() {
            if (!audioEnabled) {
                audioEnabled = true;
                [document.getElementById('clickSound'), document.getElementById('backSound')].forEach(s => {
                    s.play().then(() => { s.pause(); s.currentTime = 0; });
                });
            }
        }

        function buildAssignedCountries() {
            assignedCountries = {};
            Object.keys(contentData).forEach(fruit => {
                const a = [...contentData[fruit]];
                for (let i = a.length - 1; i > 0; i--) {
                    const j = Math.floor(Math.random() * (i + 1));
                    [a[i], a[j]] = [a[j], a[i]];
                }
                assignedCountries[fruit] = a;
            });
            localStorage.setItem('assignedCountries', JSON.stringify(assignedCountries));
        }

        function selectFruit(fruitName) {
            playClick(); enableAudio();
            currentFruit = fruitName;
            
            document.getElementById('mainView').classList.add('hidden');
            
            const loader = document.getElementById('globalLoader');
            document.getElementById('loaderText').textContent = "Sedang Memuat...";
            loader.classList.remove('hidden');
            
            setTimeout(() => {
                loader.classList.add('hidden');
                openBranchPage(fruitName);
            }, 800);
        }

        function openBranchPage(fruitName) {
            const grid = document.getElementById('branchGrid');
            grid.innerHTML = '';
            const used = usedBranches[fruitName] || [];

            branchOptions[fruitName].forEach((opt, i) => {
                const card = document.createElement('div');
                card.className = `branch-card ${used.includes(i) ? 'disabled' : ''}`;
                if (used.includes(i)) {
                    card.innerHTML = `<img src="https://ahsan.tv/wp-content/uploads/2026/05/cross.webp" class="cross-icon" alt="Sudah digunakan"><div class="branch-label">${opt.label}</div>`;
                } else {
                    card.innerHTML = `<img src="${opt.icon}" class="branch-icon" alt="${opt.label}"><div class="branch-label">${opt.label}</div>`;
                    card.onclick = () => selectBranch(i);
                }
                grid.appendChild(card);
            });

            document.getElementById('branchView').classList.remove('hidden');
            grid.classList.remove('hidden');
            grid.classList.add('pop-up');
            document.getElementById('backBtn').classList.remove('hidden');
        }

        function selectBranch(index) {
            playClick();
            document.getElementById('branchView').classList.add('hidden');
            document.getElementById('backBtn').classList.add('hidden');
            
            const loader = document.getElementById('globalLoader');
            document.getElementById('loaderText').textContent = "Mencari...";
            loader.classList.remove('hidden');

            setTimeout(() => {
                if (!usedBranches[currentFruit]) usedBranches[currentFruit] = [];
                usedBranches[currentFruit].push(index);
                localStorage.setItem('usedBranches', JSON.stringify(usedBranches));
                
                document.getElementById('answerText').textContent = assignedCountries[currentFruit][index];
                loader.classList.add('hidden');
                document.getElementById('detailView').classList.remove('hidden');
                document.getElementById('backBtn').classList.remove('hidden');
            }, 800);
        }

        /* ── Theme Switcher Logic ── */
        let themePanelOpen = false;
        const THEME_KEY = 'selectedTheme';

        function toggleThemePanel() {
            enableAudio();
            themePanelOpen = !themePanelOpen;
            const panel = document.getElementById('themePanel');
            if (themePanelOpen) {
                panel.classList.add('open');
            } else {
                panel.classList.remove('open');
            }
        }

        function applyTheme(theme) {
            playClick();
            const body = document.body;
            const skyVideo = document.getElementById('skyVideoBg');
            const desertVideo = document.getElementById('desertVideoBg');

            // Remove all theme classes
            body.classList.remove('theme-emerald', 'theme-desert');

            // Apply selected theme and control videos
            if (theme === 'emerald') {
                body.classList.add('theme-emerald');
                skyVideo.pause();
                desertVideo.pause();
            } else if (theme === 'desert') {
                body.classList.add('theme-desert');
                skyVideo.pause();
                desertVideo.play().catch(() => {});
            } else {
                // cloudy (default)
                skyVideo.play().catch(() => {});
                desertVideo.pause();
            }

            // Save to localStorage
            localStorage.setItem(THEME_KEY, theme);

            // Update active indicators
            document.getElementById('opt-cloudy').classList.toggle('active', theme === 'cloudy');
            document.getElementById('opt-emerald').classList.toggle('active', theme === 'emerald');
            document.getElementById('opt-desert').classList.toggle('active', theme === 'desert');

            // Close panel after selection
            setTimeout(() => {
                themePanelOpen = false;
                document.getElementById('themePanel').classList.remove('open');
            }, 300);
        }

        // Close theme panel when clicking outside
        document.addEventListener('click', function(e) {
            if (themePanelOpen &&
                !document.getElementById('themePanel').contains(e.target) &&
                !document.getElementById('themeBtn').contains(e.target)) {
                themePanelOpen = false;
                document.getElementById('themePanel').classList.remove('open');
            }
        });

        // Load saved theme on startup
        (function initTheme() {
            const saved = localStorage.getItem(THEME_KEY) || 'cloudy';
            applyTheme(saved);
        })();
    </script>
